 37      """ 
 38      Description 
 39       
 40          performs a global alignment of two sequences using the 
 41          Needleman-Wunsch algorithm 
 42       
 43      1st arg: seq1 
 44       
 45          a iterable sequence (string, tuple, list) 
 46           
 47      2nd arg: seq2 
 48       
 49          a iterable sequence (string, tuple, list) 
 50           
 51      3rd arg: sim_score 
 52       
 53          a function which takes two arguments - an element form seq1 and an 
 54          element from seq2 - and returns a numerical similarity score 
 55           
 56      4th arg: gap_cost 
 57       
 58          a function which takes as argument an element from seq1 or seq2 and 
 59          returns a (negative) numerical cost for skipping this element 
 60   
 61      Return value 
 62       
 63          a tuple consistig of (1) a table where the keys are (i,j) indices 
 64          of elements from seq1 and seq2, and the values are alignment scores; 
 65          (2) a list of (i,j) indices of aligned elements from seq1 and seq2 
 66           
 67      """ 
 68      # this is a straight-forward implementation -  
 69      # there is certainly room for improvement 
 70       
 71      # The symbols from the steps are: 
 72      # T = Terminate 
 73      # D = Diagonal 
 74      # U = Up 
 75      # L = Left 
 76       
 77      # create scoring matrix 
 78      scores = {(0,0): (0, "T")} 
 79       
 80      for i in range(1, len(seq1) + 1): 
 81          scores[i, 0] = (gap_cost(seq1[i-1]) * i, "U") 
 82       
 83      for j in range(1, len(seq2) + 1): 
 84          scores[0, j] = (gap_cost(seq2[j-1]) * j, "L") 
 85       
 86      for i in range(1, len(seq1) + 1): 
 87          for j in range(1, len(seq2) + 1): 
 88              scores[i, j] = max( 
 89                  (scores[i-1, j][0] + gap_cost(seq1[i-1]), "U"), 
 90                  (scores[i-1, j-1][0] + sim_score(seq1[i-1], seq2[j-1]), "D"), 
 91                  (scores[i, j-1][0] + gap_cost(seq2[j-1]), "L") ) 
 92       
 93      # backtrack to find (a) best solution 
 94      i  = len(seq1) 
 95      j  = len(seq2) 
 96      alignment = [] 
 97       
 98      while i + j > 0: 
 99          if scores[i,j][1] == "U": 
100              # Up 
101              i -= 1 
102          elif scores[i,j][1] == "D": 
103              # Diagonal 
104              # tricky: the score matrix has an initial row/column for the base 
105              # scores, which means that the actual sequence start at index 1, so we 
106              # have to subtract 1 from both i and j to return to zero-based 
107              # indexing 
108              alignment.insert(0, (i-1, j-1)) 
109              i -= 1 
110              j -= 1 
111          elif scores[i,j][1] == "L": 
112              # Left 
113              j -= 1 
114               
115      return scores, alignment
