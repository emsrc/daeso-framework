163      """ 
164      # ties are resolved according to the following two rules 
165      # (cf. eq 5 and 6 in Waterman & Eggert paper) 
166      # 
167      # (1) If score[i,j] == score[k,l] and  
168      #     i+j < k+l, 
169      #     then traceback from (i,j) 
170      #  
171      # (2) If score[i,j] == score[k,l] and 
172      #     i+j == k+l and 
173      #     i < k, 
174      #     then traceback from (i,j) 
175       
176      best_score = 0 
177      best_pair = (len(seq1) + 2, len(seq2) + 2)  
178       
179      # this might not be the shortest statement, 
180      # but certainly one the clearest 
181      for (pair, score) in scores.items(): 
182          if score[0] < best_score: 
183              continue 
184          elif score[0] == best_score: 
185              if sum(best_pair) < sum(pair): 
186                  continue 
187              elif sum(best_pair) == sum(pair): 
188                  if best_pair[0] < pair[0]: 
189                      continue 
190
