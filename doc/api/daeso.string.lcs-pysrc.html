 50      # Fill dictionary lcsLen[i,j] with length of LCS of seq1[:i] and seq2[:j] 
 51      lcsLen = {} 
 52      for i in range(len(seq1) + 1): 
 53          for j in range(len(seq2) + 1): 
 54              if i == 0 or j == 0: 
 55                  lcsLen[i,j] = 0 
 56              elif seq1[i-1] == seq2[j-1]: 
 57                  lcsLen[i,j] = lcsLen[i-1,j-1] + 1 
 58              else: 
 59                  lcsLen[i,j] = max(lcsLen[i-1,j], lcsLen[i,j-1]) 
 60   
 61      # Produce actual sequence by backtracking through pairs (i,j), 
 62      # using computed lcsLen values to guide backtracking 
 63      i = len(seq1) 
 64      j = len(seq2) 
 65      LCS = [] 
 66      while lcsLen[i,j]: 
 67          while lcsLen[i,j] == lcsLen[i-1,j]:

 69          while lcsLen[i,j] == lcsLen[i,j-1]:

 71          i -= 1 
 72          j -= 1
