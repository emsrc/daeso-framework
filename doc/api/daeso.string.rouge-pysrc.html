 90 -def rouge_l(candidate_tokens, reference_tokens, beta=1): 

 91      """ 
 92      Return the ROUGE-L score (longest common subsequence), where 
 93      candidate_tokens is a sequence of candidate tokens, 
 94      reference_tokens is a sequence of reference tokens, and 
 95      beta determines the relative importance of recall in the F-measure 
 96      """ 
 97      lcs = len(longest_common_subsequence(candidate_tokens, reference_tokens)) 
 98      recall = lcs / float(len(reference_tokens)) 
 99      precision = lcs / float(len(candidate_tokens)) 
100   
101      try: 
102          return ( ((1 + beta ** 2) * precision * recall )  / 
103                   (beta ** 2 * precision + recall) ) 
104      except ZeroDivisionError: 
105          return 0

125 -def rouge_skip_bigram(skip_bigrams1, skip_bigrams2, beta=1): 

126      """ 
127      Return the ROUGE-S score for two lists of skip bigrams, where 
128      beta determines the relative importance of recall in the F-measure 
129      """ 
130      matches = set(skip_bigrams1).intersection(set(skip_bigrams2)) 
131      # matches_count corresponds to SKIP2(X,Y) in Lin04 
132      # It is defined as "the number of skip bigram matches between X and Y"- 
133      # However, it is not clear what to do with multiple occurrences of the same bigram. 
134      # For instance, does SKIP([a,a], [a]) equal 1 or 2? Does it equal SKIP([a], [a, a])?  
135      # In the current implementation, X and Y are effectively considered sets (i.e. no duplicates).  
136      matches_count = float(len(matches)) 
137       
138      recall = matches_count / len(skip_bigrams1) 
139      precision = matches_count / len(skip_bigrams2) 
140       
141      try: 
142          return ( ((1 + beta ** 2) * precision * recall )  / 
143                   (beta ** 2 * precision + recall) ) 
144      except ZeroDivisionError:
