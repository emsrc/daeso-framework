 63      # read parallel text corpus 
 64      text_corpus = HitaextDoc(file=text_corpus_file)

 67      # read graph banks 
 68      source_bank = GraphBank(source_graphbank_file, 
 69                              graph_formats.source) 
 70      source_bank.load()

 72                              graph_formats.target)

 76      # create an empty parallel graph corpus 
 77      graph_corpus = ParallelGraphCorpus(relations=relations) 
 78       
 79      for alignment in text_corpus.alignment: 
 80          if ( alignment.get("from_tag") != focus_tags.source or  
 81               alignment.get("to_tag") != focus_tags.target ): 
 82              continue 
 83           
 84          source_tokens = _get_elem_tokens(doc_trees.source, 
 85                                           focus_tags.source,

 87          target_tokens = _get_elem_tokens(doc_trees.target,

 89                                           alignment.get("to_id")) 
 90           
 91          if len(source_tokens) > max_token_len or len(target_tokens) > max_token_len:

 96              continue 
 97

100          # the same id in the graph banks 
101          source_graph_id = alignment.get("from_id") 
102          target_graph_id = alignment.get("to_id") 
103          graphs = Pair( 
104              source_bank.get_graph(source_graph_id), 
105              target_bank.get_graph(target_graph_id)) 
106           
107          graph_pair = GraphPair(graph_banks, graphs) 
108          graph_corpus.append(graph_pair)
